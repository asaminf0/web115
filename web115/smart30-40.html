Smarter Way to Learn Chapters 30-40

Chapter 30 - toFixed()

Code an alert that displays the number rounded to 2 decimal places and converted to a string.
var num = 1.123456; alert(num.toFixed(2));
  

Chapter 31 - new Date()

Code an alert that displays the number representing the current day.
var d = new Date(); alert(d.getDay());
  

Chapter 32 - getDate()

Code an alert that displays the current year.
var d = new Date(); alert(d.getFullYear());
  

Chapter 33 - getHours()

In an alert display the hours, minutes, and seconds of the object.
var d = new Date("May 30, 2000"); alert(d.getHours() + " " + d.getMinutes() + " " + d.getSeconds());
  

Chapter 34 - setHours()

In an alert display the hours, minutes, and seconds of the object.
var now = getHours(); var nowHours = now.getHours(); now.setHours(nowHours + 1); alert(now);
  

Chapter 35 - function()

Code a function that displays an alert, "Hello world."
function greet() { alert("Hello world"); } greet();
  

Chapter 36 - parameters

call the function with parameters.
function smart36() { function displayIdiom(part1, part2) { alert(part1 + part2); } var part1=2; var part2=4; displayIdiom(part1, part2); }
  

Chapter 37 - call function

calling the function to create the message for the alert.
function smart37() { function displayIdiom(part1, part2) { alert(part1 + part2); } var part1="mor"; var part2="ning"; displayIdiom(part1, part2) }
  

Chapter 38 - call function with parameters

displays an alert with the message "Abraham Lincoln".
function makeMessage(part1, part2) { var message = part1 + part2; return message; } var firstName = "Abraham "; var lastName = "Lincoln"; alert(makeMessage(firstName, lastName));
  

Chapter 39 - switch

displays an alert with the message "You won".
var win = true; switch(win) { case true : alert("You won"); break; case false : alert("You lost"); break; }
  

Chapter 40 - switch statement

Code a single-case switch statement that tests whether the variable has been assigned the value. If so, alert "Assigned".
var num = 10; switch(num) { case 10 : alert("Assigned"); break; }
